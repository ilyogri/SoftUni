SELECT m.ModelId, m.Name,
CONCAT(AVG(DATEPART(DAYOFYEAR, j.FinishDate) - DATEPART(DAYOFYEAR, j.IssueDate))
,' days') AS [Average Service Time]
FROM Models AS m 
JOIN Jobs AS j ON j.ModelId = m.ModelId
GROUP BY m.ModelId, m.Name
ORDER BY AVG(DATEPART(DAYOFYEAR, j.FinishDate) - DATEPART(DAYOFYEAR, j.IssueDate))
